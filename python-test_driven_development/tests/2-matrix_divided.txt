Test Cases for matrix_divided

>>> matrix_divided = __import__('2-matrix_divided').matrix_divided

# Classic matrix (positives)
>>> matrix = [[1, 2, 3], [4, 5, 6]]

# Positives and negatives
>>> matrix = [[1, -2, 3], [-4, 5, -6]]

[[0.5, -1.0, 1.5], [-2.0, 2.5, -3.0]]

# Positives and negatives, divided by negative
>>> matrix = [[1, -2, 3], [-4, 5, -6]]
